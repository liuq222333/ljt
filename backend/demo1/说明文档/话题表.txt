-- 话题表
CREATE TABLE topic (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100) NOT NULL UNIQUE,
  slug VARCHAR(120) UNIQUE,
  description VARCHAR(255),
  cover_key VARCHAR(255),
  status VARCHAR(20) NOT NULL DEFAULT 'ACTIVE',
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 动态-话题关联表
CREATE TABLE community_feed_topic (
  feed_id BIGINT NOT NULL,
  topic_id BIGINT NOT NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY uk_feed_topic (feed_id, topic_id),
  CONSTRAINT fk_cft_feed FOREIGN KEY (feed_id) REFERENCES community_feed(id) ON DELETE CASCADE,
  CONSTRAINT fk_cft_topic FOREIGN KEY (topic_id) REFERENCES topic(id) ON DELETE CASCADE
);
ALTER TABLE topic ADD COLUMN feeds_count INT NOT NULL DEFAULT 0;
