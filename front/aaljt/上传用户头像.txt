async function fetchUserInfo(id: string) {
  const res = await axios.get(`${API_BASE}/api/user/getUserById`, { params: { userId: id } });
  if (res.status === 200 && res.data) {
    const user = res.data;
    currentUser.value.id = user.userId || id;
    currentUser.value.name = user.userName || currentUser.value.name;
    if (user.avatarKey) {
      const urls = buildImgUrls([user.avatarKey]); // 依赖 VITE_MINIO_BASE 或预签名
      if (urls.length) currentUser.value.avatar = urls[0];
    } else if (user.avatarUrl) {
      currentUser.value.avatar = user.avatarUrl;
    } else {
      currentUser.value.avatar = userAvatar; // 默认占位
    }
  }
}
